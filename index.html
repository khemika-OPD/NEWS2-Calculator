function finish() { let total = 0; const scale = document.querySelector('input[name="scale"]:checked').value; // ดึงค่าคะแนนจากแต่ละหมวด const rr = parseInt(document.querySelector('input[name="rr"]:checked').value); const spo2 = (scale === "1") ? parseInt(document.querySelector('input[name="spo2"]:checked').value) : parseInt(document.querySelector('input[name="spo2_2"]:checked').value); const o2 = parseInt(document.querySelector('input[name="o2"]:checked').value); const sbp = parseInt(document.querySelector('input[name="sbp"]:checked').value); const hr = parseInt(document.querySelector('input[name="hr"]:checked').value); const loc = parseInt(document.querySelector('input[name="loc"]:checked').value); const t = parseInt(document.querySelector('input[name="t"]:checked').value); total = rr + spo2 + o2 + sbp + hr + loc + t; // ตรวจสอบข้อที่ได้คะแนน 3 (RED Score) let redCategories = []; if (rr === 3) redCategories.push("อัตราการหายใจ"); if (spo2 === 3) redCategories.push("ออกซิเจนในเลือด (SpO2)"); if (sbp === 3) redCategories.push("ความดันโลหิต (SBP)"); if (hr === 3) redCategories.push("อัตราการเต้นของหัวใจ"); if (loc === 3) redCategories.push("ระดับความรู้สึกตัว"); if (t === 3) redCategories.push("อุณหภูมิร่างกาย"); // แสดงคะแนนรวม document.getElementById('totalScore').innerText = total; const box = document.getElementById('adviceBox'); // สร้างข้อความเตือนสีแดง ถ้ามี RED Score let redAlertHtml = ""; if (redCategories.length > 0) { redAlertHtml = `
พบ RED Score ในหมวด:
${redCategories.join(", ")}

• แจ้งพยาบาลทันทีเพื่อพิจารณาส่งต่อห้องฉุกเฉิน ER
• ต้องติดตามอาการอย่างใกล้ชิด
`; } // กำหนดสีและข้อความตามคะแนนรวม if (total === 0) { box.style.backgroundColor = "#e8f5e9"; box.style.color = "#2e7d32"; box.innerHTML = `Non Urgent ไม่เร่งด่วน — ให้การดูแล ตรวจสอบอาการทั่วไป แนะนำขั้นตอนการรับบริการ`; } else if (total <= 4) { box.style.backgroundColor = "#fff9c4"; box.style.color = "#f57f17"; box.innerHTML = `Less Urgent เร่งด่วนน้อย — รายงานพยาบาลเพื่อประเมินซ้ำ พิจารณาให้พบแพทย์ภายใน 30 นาที` + redAlertHtml; } else if (total <= 6) { box.style.backgroundColor = "#fff3e0"; box.style.color = "#ef6c00"; box.innerHTML = `Urgent เร่งด่วน — แจ้งพยาบาลเพื่อประเมินอาการซ้ำและส่งต่อห้องฉุกเฉิน ER` + redAlertHtml; } else { box.style.backgroundColor = "#ffebee"; box.style.color = "#c62828"; box.innerHTML = `Emergent ฉุกเฉิน — แจ้งพยาบาลและแพทย์เพื่อส่งต่อผู้ป่วยให้ได้รับการดูแลขั้นวิกฤต` + redAlertHtml; } document.getElementById(`step${currentStep}`).classList.remove('active'); document.getElementById('resultStep').classList.add('active'); }
